Mysql的三种安装方式

mysql安装有三种方式，分别是yum安装、二进制预编译安装、源码包安装。

第一种方式：yum安装

1、安装

wget https://dev.mysql.com/get/mysql57-community-release-el7-9.noarch.rpm
rpm -ivh mysql57-community-release-el7-9.noarch.rpm
yum install mysql-server
systemctl start mysqld

2、修改密码，在日志文件中先找到初始密码（这个密码是临时的，登录mysql后必须修改才能进一步操作）

grep 'temporary password' /var/log/mysqld.log

2020-04-11T07:53:15.951160Z 1 [Note] A temporary password is generated for root@localhost: g(wfb5Nhf.2L

登录修改密码

mysqladmin -u root -p'g(wfb5Nhf.2L' password '123456'；

第二种：二进制预编译安装

1、首先去mysql官网下载对应的安装包，官网下载地址： https://downloads.mysql.com/archives/community/ Mysql5.7预编译包下载

2、下载后上传到服务器并解压到相应的目录，并修改名字

tar -zxvf mysql-5.7.33-linux-glibc2.12-x86_64.tar.gz -C /software/
cd /software/
mv mysql-5.7.33-linux-glibc2.12-x86_64/ mysql-5.7

3、创建mysql用户

useradd -r mysql -s /sbin/nologin

4、创建数据目录

mkdir -p /data/mysql/data

5、修改权限

chown -R mysql:mysql /data/mysql
chmod 750 /data/mysql

6、创建配置文件 vim /etc/my.cnf

[client]
port        = 3306
socket      = /tmp/mysql.sock

[mysqld]
#端口号
port        = 3306
#socket套接字文件
socket      = /tmp/mysql.sock
#指定用户运行服务
user = mysql

#软件安装目录
basedir = /usr/local/mysql
#数据存储目录
datadir = /data/mysql/data
#pid进程号文件
pid-file = /data/mysql/mysql.pid
#错误日志存储位置
log_error = /data/mysql/mysql-error.log
#开启慢查询日志存储
slow_query_log = 1
#设置慢查询时间
long_query_time = 1
#设置慢查询日志存储位置
slow_query_log_file = /data/mysql/mysql-slow.log

#表示启用二进制日志记录，服务器记录了所有改变数据语句的二进制日志，用于备份和复制。
log-bin=/data/mysql/binlog/mysqlbin
#指定二进制日志文件存储格式
binlog_format=ROW
#当每进行1次事务提交之后，MySQL将进行一次fsync之类的磁盘同步指令来将binlog_cache中的数据强制写入磁盘。
sync_binlog=1

#设定北京时间
default-time-zone='+08:00'

#skip-external-locking

#索引缓存大小。MyISAM表用于索引块的缓冲区大小，可以被所有线程所共享，默认值是8388608（即8M），它是对MyISAM表性能影响最大的一个参数，如果数据库存储引擎为InnoDB类型，这个参数是无效的。可以通过增大这个值，以便索引更好的处理所有读和多写操作。
key_buffer_size = 32M
#限制Server接受的数据包大小。有时候大的插入和更新会受 max_allowed_packet 参数限制，导致写入或者更新失败。最大值1G。
max_allowed_packet = 1024M
#指定表高速缓存的大小
table_open_cache = 128

net_buffer_length = 8K
read_buffer_size = 768K
read_rnd_buffer_size = 512K
myisam_sort_buffer_size = 8M
thread_cache_size = 16
query_cache_size = 16M
tmp_table_size = 32M
performance_schema_max_table_instances = 1000

explicit_defaults_for_timestamp = true
#skip-networking
max_connections = 500
max_connect_errors = 100
open_files_limit = 65535

#msyql服务器的唯一标识
server_id   = 1
#自动过期清理binlog
expire_logs_days = 10
early-plugin-load = ""

#指定
default_storage_engine = InnoDB
innodb_file_per_table = 1
innodb_buffer_pool_size = 128M
innodb_log_file_size = 32M
innodb_log_buffer_size = 8M
innodb_flush_log_at_trx_commit = 1
innodb_lock_wait_timeout = 50


[mysqldump]
quick
max_allowed_packet = 16M

[mysql]
no-auto-rehash

[myisamchk]
key_buffer_size = 32M
sort_buffer_size = 768K
read_buffer = 2M
write_buffer = 2M

7、初始化数据库

bin/mysqld --initialize --user=mysql --basedir=/data/mysql/ --datadir=/data/mysql/data

8、获取mysql初始密码

cat mysqld.log |grep password|awk -F"localhost: " '{print $2}'

9、拷贝mysql启动脚本（这里要强调下，因为mysql的启动脚本里面没有指定mysql的安装目录和数据目录，默认就是/usr/local/mysql，但是我们这里的安装目录是/software/mysql-5.7，所以我们要修改启动脚本的basedir和datadir才可以）

cp support-files/mysql.server /etc/init.d/mysqld

修改服务脚本的basedir和datadir的路径为自己安装的路径

vim /etc/init.d/mysqld

basedir=/software/mysql-5.7
datadir=/software/mysql-5.7/data

10、将mysql的启动脚本加入开机自启服务中

chkconfig --add mysqld
chkconfig mysqld on

11、启动MySQL服务

systemctl start mysqld

12、将mysql命令加入系统环境变量中

echo "PATH=/data/mysql-5/bin:$PATH">>/etc/profile
source  /etc/profile

后面和yum的配置一样，需要修改初始密码,这里改为123456 alter user user() identified by "w1ses0ft"; 最后将mysql的root设置为可以通过本地进行远程连接访问。update user set authentication_string=password("新密码") where user="root";

Mysql -uroot -p123456
use mysql;
update mysql.user set host='%' where user='root';
FLUSH privileges; 
exit;

第三种方式：源码包编译安装

安装依赖包

yum -y install gcc gcc-c++ ncurses ncurses-devel cmake

下载源码包，解压到指定目录解压到/usr/local目录下

 wget http://downloads.sourceforge.net/project/boost/boost/1.59.0/boost_1_59_0.tar.gz 
 tar -zxvf boost_1_59_0.tar.gz -C /usr/local

下载mysql5.7源码包

wget https://downloads.mysql.com/archives/get/p/23/file/mysql-5.7.32.tar.gz

添加mysql用户

useradd -r mysql -s /sbin/nologin

编译之前要确保安装目录、数据目录、配置文件目录都存在。

mkdir -p  /software/mysql-5.7/data
mkdir -p  /software/mysql-5.7/etc

编译

cmake \
-DCMAKE_INSTALL_PREFIX=/software/mysql-5.7 \
-DMYSQL_DATADIR=/software/mysql-5.7/data \
-DSYSCONFDIR=/software/mysql-5.7/etc \
-DWITH_BOOST=/usr/local/boost_1_59_0 \
-DWITH_INNOBASE_STORAGE_ENGINE=1 \
-DWITH_PARTITION_STORAGE_ENGINE=1 \
-DWITH_FEDERATED_STORAGE_ENGINE=1 \
-DWITH_BLACKHOLE_STORAGE_ENGINE=1 \
-DWITH_MYISAM_STORAGE_ENGINE=1 \
-DENABLED_LOCAL_INFILE=1 \
-DENABLE_DTRACE=0 \
-DDEFAULT_CHARSET=utf8mb4 \
-DDEFAULT_COLLATION=utf8mb4_general_ci \
-DWITH_EMBEDDED_SERVER=1

-DDOWNLOAD_BOOST=1 加上这个选项会在编译的时候自动下载boost 编译出错，删除下面文件，重新编译安装。 #rm -f CMakeCache.txt 如果echo $?返回0说明预编译成功，那么继续编译安装 make && make install

拷贝启动脚本，设置开机自启动

cp support-files/mysql.server /etc/init.d/mysqld

#修改服务脚本的basedir和datadir的路径为自己安装的路径 vim /etc/init.d/mysqld basedir=/software/mysql-5.7 datadir=/software/mysql-5.7/data

chmod +x /etc/init.d/mysqld
chkconfig --add mysqld
chkconfig mysqld on

在/software/mysql-5.7/etc目录创建配置文件my.cnf；参数见上方yum安装方式或预编译安装。

启动MySQL服务

 systemctl start mysqld

12、将mysql命令加入系统环境变量中

echo "PATH=/da/mysql/bin:$PATH">>/etc/profile
source  /etc/profile

初始化数据库

bin/mysqld --initialize --user=mysql --basedir=/da/mysql/ --datadir=/data/mysql/data
alter user 'root@'%' identified with mysql_native_password by 'w1ses0ft';

修改服务脚本的basedir和datadir的路径为自己安装的路径

vim /etc/init.d/mysqld

basedir=/data/mysql
datadir=/data/mysql/data

chkconfig --add mysqld
